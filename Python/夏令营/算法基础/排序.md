

## 归并排序

```python
# 归并排序
n = int(input())
nums = list(map(int, input().split()))
def sort(l, r):
    if l == r: return
    m = l + (r - l) // 2
    sort(l, m)
    sort(m + 1, r)
    res = []
    p1, p2 = l, m + 1
    while p1 <= m or p2 <= r:
        if p1 > m:
            res.append(nums[p2])
            p2 += 1
        elif p2 > r:
            res.append(nums[p1])
            p1 += 1
        elif nums[p1] <= nums[p2]:
            res.append(nums[p1])
            p1 += 1
        else:
            res.append(nums[p2])
            p2 += 1
    for i in range(l, r + 1):
        nums[i] = res[i - l]
sort(0, len(nums) - 1)
print(*nums)
```
